1. indexOf / lastIndexOf

arr = ["사과", "복숭아", "포도", "파인애플","사과", "메론", "배", "자두","방울토마토"]
indexOf("사과") = 사과가 몇번째 있는지 indexOf("사과",3) = 3번째 인덱스 이후 사과가 몇번째 있는지 lastIndexOf("사과") = 사과가 뒤에서 몇번째 있는지
